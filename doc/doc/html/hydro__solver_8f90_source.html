121 subroutine step(dt)
122 
123  use parameters, only : nx, ny, nz, neqdyn
124 #ifdef CT
125  use globals, only: up, u, primit, f, g, h, dx, dy, dz, e
126 #else
127  use globals, only : up, u, primit, f, g, h, dx, dy, dz
128 #endif
129 #if defined(GRAV) || defined(RADPRES) || defined(EIGHT_WAVE)
130  use sources
131 #endif
132  implicit none
133 #if defined(GRAV) || defined(RADPRES) || defined(EIGHT_WAVE)
134  real :: s(neq)
135 #endif
136  real, intent(in) :: dt
137  integer :: i, j, k
138  real :: dtdx, dtdy, dtdz
139 
140  dtdx=dt/dx
141  dtdy=dt/dy
142  dtdz=dt/dz
143 
144  do k=1,nz
145  do j=1,ny
146  do i=1,nx
147 
148  up(:5,i,j,k)=u(:5,i,j,k)-dtdx*(f(:5,i,j,k)-f(:5,i-1,j,k)) &
149  -dtdy*(g(:5,i,j,k)-g(:5,i,j-1,k)) &
150  -dtdz*(h(:5,i,j,k)-h(:5,i,j,k-1))
151 #ifdef MHD
152 #ifdef CT
153  up(6,i,j,k)=u(6,i,j,k)-0.5*dtdy*(e(3,i,j+1,k)-e(3,i,j-1,k)) &
154  +0.5*dtdz*(e(2,i,j,k+1)-e(2,i,j,k-1))
155 
156  up(7,i,j,k)=u(7,i,j,k)+0.5*dtdx*(e(3,i+1,j,k)-e(3,i-1,j,k)) &
157  -0.5*dtdz*(e(1,i,j,k+1)-e(1,i,j,k-1))
158 
159  up(8,i,j,k)=u(8,i,j,k)-0.5*dtdx*(e(2,i+1,j,k)-e(2,i-1,j,k)) &
160  +0.5*dtdy*(e(1,i,j+1,k)-e(1,i,j-1,k))
161 
162 #else
163  up(6:8,i,j,k)=u(6:8,i,j,k)-dtdx*(f(6:8,i,j,k)-f(6:8,i-1,j,k)) &
164  -dtdy*(g(6:8,i,j,k)-g(6:8,i,j-1,k)) &
165  -dtdz*(h(6:8,i,j,k)-h(6:8,i,j,k-1))
166 #endif
167 #endif
168 
169 #ifdef PASSIVES
170  up(neqdyn+1:,i,j,k)=u(neqdyn+1:,i,j,k)-dtdx*(f(neqdyn+1:,i,j,k)-f(neqdyn+1:,i-1,j,k)) &
171  -dtdy*(g(neqdyn+1:,i,j,k)-g(neqdyn+1:,i,j-1,k)) &
172  -dtdz*(h(neqdyn+1:,i,j,k)-h(neqdyn+1:,i,j,k-1))
173 #endif
174 #if defined(GRAV) || defined(RADPRES) || defined(EIGHT_WAVE)
175  call source(i,j,k,primit(:,i,j,k),s)
176  up(:,i,j,k)= up(:,i,j,k)+dt*s(:)
177 #endif
178 
179  end do
180  end do
181  end do

183 end subroutine step

190 subroutine tstep()
191 
192  use parameters, only : tsc
193  use globals
194  use hydro_core, only : calcprim
195  use boundaries
196 #ifdef C2ray
197  use c2ray_rt, only: c2ray_radiative_transfer
198 #endif
199 #ifdef COOLINGH
200  use cooling_h
201 #endif
202 #ifdef COOLINGDMC
203  use cooling_dmc
204 #endif
205 #ifdef COOLINGCHI
206  use cooling_chi
207 #endif
208 #ifdef RADDIFF
209  use difrad
210 #endif
211 
212 #ifdef THERMAL_COND
213  use thermal_cond
214 #endif
215  implicit none
216  real :: dtm
217 
218  ! 1st half timestep ========================
219  dtm=dt_cfl/2.
220  ! calculate the fluxes using the primitives
221  ! (piecewise constant)
222 #ifdef HLL
223  call hllfluxes(1)
224 #endif
225 #ifdef HLLC
226  call hllcfluxes(1)
227 #endif
228 #ifdef HLLE
229  call hllefluxes(1)
230 #endif
231 #ifdef HLLD
232  call hlldfluxes(1)
233 #endif
234 
235  !calculates the electric current for CT
236 #ifdef CT
237  call current()
238 #endif
239  ! upwind timestep
240  call step(dtm)
241 
242  ! add viscosity
243  !call viscosity()
244 
245  ! 2nd half timestep ========================
246  ! boundaries in up and primitives up ---> primit
247  call boundaryii()
248  call calcprim(up,primit)
249 
250  ! calculate the fluxes using the primitives
251  ! with linear reconstruction (piecewise linear)
252 #ifdef HLL
253  call hllfluxes(2)
254 #endif
255 #ifdef HLLC
256  call hllcfluxes(2)
257 #endif
258 #ifdef HLLE
259  call hllefluxes(2)
260 #endif
261 #ifdef HLLD
262  call hlldfluxes(2)
263 #endif
264 
265  !calculates the electric current for CT
266 #ifdef CT
267  call current()
268 #endif
269  ! upwind timestep
270  call step(dt_cfl)
271 
272  ! add viscosity
273  call viscosity()
274 
275  ! copy the up's on the u's
276  u=up
277 
278  ! update the chemistry network
279  ! at this point is in cgs
280 #ifdef EOS_CHEM
281  ! the primitives in the physical celles are upated
282  call update_chem()
283 #endif
284 
285  ! Do the Radiaiton transfer (Monte Carlo type)
286 #ifdef RADDIFF
287  call diffuse_rad()
288 #endif
289 
290 #ifdef CEXCHANGE
291  !****************************************************
292  ! apply charge exchange TO BE ADDED IN COLLING?
293  ! not fully implemented
294  !****************************************************
295  call cxchange(dt*tsc)
296 #endif
297 
298  ! apply cooling/heating
299 #ifdef COOLINGH
300  ! add cooling to the conserved variables
301  call coolingh()
302  ! update the primitives with u
303  call calcprim(u, primit)
304 #endif
305 #ifdef COOLINGDMC
306  ! the primitives are updated in the cooling routine
307  call coolingdmc()
308 #endif
309 #ifdef COOLINGCHI
310  ! the primitives are updated in the cooling routine
311  call coolingchi()
312 #endif
313 #ifdef COOLINGCHEM
314  !the primitives are already updated in update_chem
315  call cooling_chem()
316 #endif
317 
318 #ifdef C2ray
319  ! Apply Rad transfer, and heating & Cooling w/C2-Ray
320  call c2ray_radiative_transfer(dt_cfl*tsc)
321 #endif
322 
323  ! boundary contiditions on u
324  call boundaryi()
325 
326  ! update primitives on the boundaries
327  call calcprim(u,primit,only_ghost=.true.)
328 
329 
330 #ifdef THERMAL_COND
331  ! Thermal conduction
332  call thermal_conduction()
333 #endif
334 
335 end subroutine tstep
