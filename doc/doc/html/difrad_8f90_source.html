148 subroutine random_versor(xd,yd,zd)

150  implicit none

152  real :: r2, w
153  real :: ran(3)

157  call random_number(ran)
158  xd=2.*(ran(1)-0.5)

186 subroutine starsource(srad,x0,y0,z0,x,y,z,xd,yd,zd)
187  implicit none
188  real, intent (in) :: Srad, x0, y0, z0
189  real, intent (out) :: x, y, z, xd, yd, zd
190  real :: xs, ys,zs
191  real :: rd, rs, w, prod
192  real :: ran(3)
193 
194  ! get unit vector from star center
195  rs=2.
196  do while(rs > 1.)
197  call random_number(ran)
198  xs=2.*(ran(1)-0.5)
199  ys=2.*(ran(2)-0.5)
200  zs=2.*(ran(3)-0.5)
201  rs= xs**2 + ys**2 + zs**2
202  end do
203  w=srad/sqrt(rs)
204  xs=xs*w
205  ys=ys*w
206  zs=zs*w
207 
208  ! the position of the source
209  x=x0+xs
210  y=y0+ys
211  z=z0+zs
212 
213  ! get the direction of the ray
214  rd=2.
215  do while(rd > 1.)
216  call random_number(ran)
217  xd=2.*(ran(1)-0.5)
218  yd=2.*(ran(2)-0.5)
219  zd=2.*(ran(3)-0.5)
220  rd = xd**2 + yd**2 + zd**2
221  end do
222  w=1./sqrt(rd)
223  xd=xd*w
224  yd=yd*w
225  zd=zd*w
226 
227  ! reflect rays that point to the star
228  prod=( xs*xd+ ys*yd + zs*zd )
229  if (prod < 0. ) then
230  xd=-xd
231  yd=-yd
232  zd=-zd
233  endif
234 
235 end subroutine starsource

347

349  .and. (i <= nx).and.(i >= 1) &

351  .and. (k <= nz).and.(k >= 1) &
