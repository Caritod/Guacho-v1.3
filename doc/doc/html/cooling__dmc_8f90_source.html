122 
123  if(t > tmin) then
124

126 
127  aloss=cooldmc(t)
128  dens=primit(1,i,j,k)
129  ce=(aloss*dble(dens)**2)/(eth0*psc) ! cgs
130 
131  ! apply cooling to primitive and conserved variables
132  primit(5,i,j,k)=primit(5,i,j,k)*exp(-ce*dt_seconds)
133

135  u(5,i,j,k)=u(5,i,j,k)-eth0+cv*primit(5,i,j,k)
