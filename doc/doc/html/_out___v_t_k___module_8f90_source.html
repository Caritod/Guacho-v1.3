172  real(primit(4,i,j,k)*vsc,4)

183  do j=1,ny
184  do i=1,nx

186  real(primit(7,i,j,k)*bsc,4), &
187  real(primit(8,i,j,k)*bsc,4)
188  end do
189  end do
190  end do
191  write(unitout) lf
192 
193 #endif
194
