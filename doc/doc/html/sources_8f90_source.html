81 subroutine grav_source(xc,yc,zc,pp,s)
82  use constants, only : ggrav
83  use exoplanet ! this module contains the position of the planet
84  implicit none
85  real, intent(in) :: xc, yc, zc
86  real, intent(in) :: pp(neq)
87  real, intent(inout) :: s(neq)
88  integer, parameter :: nb=2 ! 2 particles
89  real :: x(nb),y(nb),z(nb), GM(nb), rad2(nb)
90  integer :: i
91 
92  gm(1)=0.3*ggrav*masss/rsc/vsc2
93  gm(2)=ggrav*massp/rsc/vsc2
94 
95  !calculate distance from the sources
96  ! star
97  x(1)=xc
98  y(1)=yc
99  z(1)=zc
100  rad2(1) = x(1)**2 +y(1)**2 + z(1)**2
101  ! planet
102  x(2)=xc-xp
103  y(2)=yc
104  z(2)=zc-zp
105  rad2(2) = x(2)**2 +y(2)**2 + z(2)**2
106 
107  ! update source terms
108  do i=1, nb
109  ! momenta
110  s(2)= s(2)-pp(1)*gm(i)*x(i)/(rad2(i)**1.5)
111  s(3)= s(3)-pp(1)*gm(i)*y(i)/(rad2(i)**1.5)
112  s(4)= s(4)-pp(1)*gm(i)*z(i)/(rad2(i)**1.5)
113  ! energy
114  s(5)= s(5)-pp(1)*gm(i)*( pp(2)*x(i) +pp(3)*y(i) +pp(4)*z(i) ) &
115  /(rad2(i)**1.5 )
116  end do
117 
118 
119 end subroutine grav_source
