45 subroutine ludcmp(a, n, indx,d)
46 
47  implicit none
48  integer, intent(in) :: n
49  real (kind=8), intent(inout) :: a(n,n), d
50  integer, intent(out) :: indx(n)
51 
52  real (kind=8) :: vv(n*n), aamax, dum, sum
53  real (kind=8), parameter :: tiny=1.0e-25
54  integer :: i, imax, j, k
55 
56  d=1.
57  do i=1,n
58  aamax=0.
59  do j=1,n
60  if(abs(a(i,j)).gt.aamax) aamax=abs(a(i,j))
61  end do
62  ! if(aamax.eq.0.) pause ' singular matrix.'
63  vv(i)=1./aamax
64  end do
65 
66  do j=1,n
67  if(j.gt.1) then
68  do i=1,j-1
69  sum=a(i,j)
70  if(i.gt.1) then
71  do k=1,i-1
72  sum=sum-a(i,k)*a(k,j)
73  end do
74  a(i,j)=sum
75  end if
76  end do
77  end if
78  aamax=0.
79  do i=j,n
80  sum=a(i,j)
81  if(j.gt.1) then
82  do k=1,j-1
83  sum=sum-a(i,k)*a(k,j)
84  end do
85  a(i,j)=sum
86  end if
87  dum=vv(i)*abs(sum)
88  if(dum.ge.aamax) then
89  imax=i
90  aamax=dum
91  end if
92  end do
93  if(j.ne.imax) then
94  do k=1,n
95  dum=a(imax,k)
96  a(imax,k)=a(j,k)
97  a(j,k)=dum
98  end do
99  d=-d
100  vv(imax)=vv(j)
101  end if
102  indx(j)=imax
103  if(j.ne.n) then
104  if(a(j,j).eq.0.) a(j,j)=tiny
105  dum=1./a(j,j)
106  do i=j+1,n
107  a(i,j)=a(i,j)*dum
108  end do
109  end if
110  end do
111  if(a(n,n).eq.0.) a(n,n)=tiny
112  return
113 
114 end subroutine ludcmp

128 subroutine lubksb(a,n,indx,b)

135  integer :: i, ii, j, ll

137 
138  ii=0

142  b(ll)=b(i)
143  if(ii.ne.0) then
144  do j=ii,i-1

146  end do

148  ii=i
149  end if

151  end do
152  do i=n,1,-1

154  if(i.lt.n) then
155  do j=i+1,n
156  sum=sum-a(i,j)*b(j)
157  end do
158  end if

160  end do
