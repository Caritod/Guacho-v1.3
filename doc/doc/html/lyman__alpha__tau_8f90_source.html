283 subroutine fill_map(nxmap,nymap,nvmap,vmin,vmax,u,map,dxT,dyT,&
284  theta_x,theta_y,theta_z)
285 
286  use constants, only : clight
287  use parameters, only : nxmin, nxmax, nymin, nymax, nzmin, nzmax, &
288  neq, nx, ny, nz, vsc2, rsc,nztot, neqdyn
289  use globals, only : dz
290  use hydro_core, only : u2prim

293 
294  integer, intent(in) :: nxmap,nymap,nvmap
295  real, intent(in) :: vmin, vmax
296  real, intent(in) :: u(neq,nxmin:nxmax,nymin:nymax, nzmin:nzmax)
297  real , intent(in) :: dxT, dyT, theta_x, theta_y, theta_z
298  real, intent(out) :: map(nxmap,nymap,nvmap)
299  integer :: i,j,k, iobs, jobs
300  real :: x,y,z,xn,yn,zn, vx, vy, vz,vxn, vyn, vzn, velsc
301  real :: T, prim(neq), profile(nvmap)
302  real, parameter :: sigmaLA = 0.01105, lambdala=1.215668e-5 !(c/nu0=lambda)
303  velsc=sqrt(vsc2)
304 
305  do k=1,nz
306  do j=1,ny
307  do i=1,nx
308 
309  ! obtain original position
310  call getxyz(i,j,k, x,y,z)
311  !only do to the side facing the observer

320  jobs=yn/dyt + nymap/2
321 
322  ! get the velocity in cm/s

326  vz=prim(4)*velsc

328  call rotation_x(theta_x,vx,vy,vz,vxn,vyn,vzn)
329  call rotation_y(theta_y,vxn,vyn,vzn,vx,vy,vz)
330  call rotation_z(theta_z,vx,vy,vz,vxn,vyn,vzn)
331 
332  ! calculate the line profile function
333  call phigauss(t, vzn,vmin,vmax,nvmap,profile)
334  ! make sure the result lies in the map bounds
335  if( (iobs >=1 ).and.(jobs >=1 ).and. &
336  (iobs <=nxmap).and.(jobs <=nymap) ) then
337  if ((t < 1e5).and.(prim(7)<0)) then
338  map(iobs,jobs,:)= map(iobs,jobs,:) + &
339  dz*rsc*prim(neqdyn+1)*sigmala*lambdala*profile(:)
340  end if
341  end if
342  end if
343  end do
344  end do
345  end do
346 
347 end subroutine fill_map

385 subroutine phigauss(T,vzn,vmin,vmax,nvmap,profile)
386 
387  use constants, only: amh, pi, kb, clight
388  implicit none
389  real, intent(in) :: T, vzn, vmin, vmax
390  integer, intent(in) :: nvmap
391  real, intent(out) :: profile(nvmap)
392  integer :: i
393  real :: coef, dv, vr
394 
395  profile(:)=0.
396  dv=(vmax-vmin)/float(nvmap)
397 
398  coef=amh/(2.*kb*t)
399 
400  do i=1,nvmap
401  vr=(float(i)-0.5)*dv+vmin
402  profile(i)=sqrt(coef/pi)*exp(-coef*((vr-vzn)**2) )
403  end do
404
