48 subroutine prim2fhlle(priml,primr,ff)
49 
50  use parameters, only : neq
51  use hydro_core, only : cfastx, prim2f, prim2u
52  implicit none
53  real, dimension(neq),intent(in ) :: priml, primr
54  real, dimension(neq),intent(inout) :: ff
55  real, dimension(neq) :: uR, uL, fL, fR
56  real :: csl, csr, sl, sr
57 
58  call cfastx(priml,csl)
59  call cfastx(primr,csr)
60 
61  sr=max(priml(2)+csl,primr(2)+csr)
62  sl=min(priml(2)-csl,primr(2)-csr)
63 
64  if (sl > 0) then
65  call prim2f(priml,ff)
66  return
67  endif
68 
69  if (sr < 0) then
70  call prim2f(primr,ff)
71  return
72  endif
73 
74  call prim2f(priml,fl)
75  call prim2f(primr,fr)
76  call prim2u(priml,ul)
77  call prim2u(primr,ur)
78 
79  ff(:)=(sr*fl(:)-sl*fr(:)+sl*sr*(ur(:)-ul(:)))/(sr-sl)
80 
81  end subroutine prim2fhlle

93 subroutine hllefluxes(choice)

102  !

122  call swapy(primr,neq) !swaps primR for R state
123 
124  call prim2fhlle(priml,primr,ff) !gets fluxes (swapped)
125  call swapy(ff,neq) !swaps back the fluxes
126  g(:,i,j,k)=ff(:)
127

131  call swapz(priml,neq)

134  call prim2fhlle(priml,primr,ff)
135  call swapz(ff,neq)
136  h(:,i,j,k)=ff(:)
137 
138  end do
139  end do

141

143 
144  do k=0,nz
145  do j=0,ny
146  do i=0,nx
147 
148  !------- x direction ------------------------------------
149  priml(:)=primit(:,i, j,k )

151  primll(:)=primit(:,i-1,j,k )

160  primr(:)=primit(:,i,j+1,k )

162  primrr(:)=primit(:,i,j+2,k )

165  call swapy(primll,neq)
166  call swapy(primrr,neq)

168 
169  call prim2fhlle(priml,primr,ff)
