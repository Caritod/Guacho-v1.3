57 subroutine initial_conditions(u,time)
58 
59  use parameters, only : neq, nxmin, nxmax, nymin, nymax, nzmin, nzmax
60  use globals, only : coords, dx, dy, dz, rank
61  implicit none
62  real, intent(out) :: u(neq,nxmin:nxmax,nymin:nymax,nzmin:nzmax)
63  real, intent(in) :: time
64 
65  integer :: i,j,k
66  real :: x,y,z, rads, velx, vely, velz, dens
67  ! the star wind does not cover the entire domain, we fill here
68  ! as if the exoplanet is absent
69  do i=nxmin,nxmax
70  do j=nymin,nymax
71  do k=nzmin,nzmax
72 
73  ! Position measured from the centre of the grid (star)
74  x=(float(i+coords(0)*nx-nxtot/2)+0.5)*dx
75  y=(float(j+coords(1)*ny-nytot/2)+0.5)*dy
76  z=(float(k+coords(2)*nz-nztot/2)+0.5)*dz
77 
78  ! Distance from the centre of the star
79  rads=sqrt(x**2+y**2+z**2)
80 
81  velx=vsw*x/rads
82  vely=vsw*y/rads
83  velz=vsw*z/rads
84  dens=dsw*rsw**2/rads**2
85  ! total density and momena
86  u(1,i,j,k) = dens
87  u(2,i,j,k) = dens*velx
88  u(3,i,j,k) = dens*vely
89  u(4,i,j,k) = dens*velz
90  ! total energy
91  u(5,i,j,k)=0.5*dens*(velx**2+vely**2+velz**2) &
92  + cv*dens*1.9999*tsw
93 
94  u(neqdyn+1,i,j,k)= 0.0001*dens ! density of neutrals
95  u(neqdyn+2,i,j,k)= dens ! passive scalar
96 
97  end do
98  end do
99  end do
100 
101  call impose_exo(u,0.)
102 
103 end subroutine initial_conditions
