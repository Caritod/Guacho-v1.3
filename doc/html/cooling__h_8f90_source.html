159 FUNCTION aloss(X1,X2,DT,DEN,DH0,TE0)
160 
161  implicit none
162 
163  real (kind=8) :: ALOSS
164  real, intent(in) :: DT
165  !real (kind=8), intent(in) :: X1,X2,DEN,DH0,TE0
166  real (kind=8), intent(in) :: X1,X2,DEN,DH0,TE0
167  real, parameter :: XION=2.179e-11, xh=0.9,xo=1.e-3
168  real, parameter :: C0=0.5732,c1=1.8288e-5,c2=-1.15822e-10,c3=9.4288e-16
169  real, parameter :: D0=0.5856,d1=1.55083e-5,d2=-9.669e-12, d3=5.716e-19
170  real, parameter :: ENK=118409.,en=1.634e-11
171  real (kind=8) :: SHP
172  !real (kind=8) :: TE, DH,DHP,DE,DOI,DOII,OMEGA,OMEGAL,OMEGAH,FRAC,QLA
173  !real (kind=8) :: ECOLL,CION,EION,EREC,TM,T2,EOI,EOII,EQUIL,FR,EX2,TANH
174  !real (kind=8) :: BETAH, BETAF, HIICOOL
175  real (kind=8) :: TE, DH,DHP,DE,DOI,DOII,OMEGA,OMEGAL,OMEGAH,FRAC,QLA
176  real (kind=8) :: ECOLL,CION,EION,EREC,TM,T2,EOI,EOII,EQUIL,FR,EX2,TANH
177  real (kind=8) :: BETAF, HIICOOL!, BETAH
178 
179  te=max(te0,10.)
180  dh=den
181  dhp=(1.-x1)*dh
182  de=dhp+1.e-4*dh
183  doi=xo*dh0
184  doii=xo*dhp
185 
186  shp=-dh*(x1-x2)/dt ! ?```
187  if(te <= 1e4 ) then
188  aloss = 0.
189  return
190  end if
191 
192  ! Collisionally excited Lyman alpha
193  IF(te.LE.55000.) omega=c0+te*(c1+te*(c2+te*c3))
194  IF(te.GE.72000.) omega=d0+te*(d1+te*(d2+te*d3))
195  IF(te.GT.55000..AND.te.LT.72000.) THEN
196  omegal=c0+te*(c1+te*(c2+te*c3))
197  omegah=d0+te*(d1+te*(d2+te*d3))
198  frac=(te-55000.)/17000.
199  omega=(1.-frac)*omegal+frac*omegah
200  END IF
201  qla=8.6287e-6/(2.*sqrt(te))*omega*exp(-enk/te)
202  ecoll=de*dh0*qla*en
203  ecoll=max(ecoll,0.)
204 
205  ! Hydrogen recombination and collisional ionization
206 
207  cion=5.834e-11*sqrt(te)*exp(-1.579e5/te)
208  ! AREC=2.61E-10/TE**0.7
209  ! EREC=DE*DHP*(BETAH(TE)+AREC*XION)
210  ! EION=SHP*XION
211  eion=de*dh0*cion*xion
212  ! EION=0.
213  erec=de*dhp*(betah(te))
214  erec=max(erec,0.)
215 
216  ! [O I] and [O II] coll. excited lines
217 
218  tm=1./te
219  t2=tm*tm
220  eoi=de*doi*10.**(1381465*t2-12328.69*tm-19.82621)
221  eoii=de*doii*10.**(-2061075.*t2-14596.24*tm-19.01402)
222  eoi=max(eoi,0.)
223  eoii=max(eoii,0.)
224 
225  ! free-free cooling
226 
227  betaf=1.3*1.42e-27*te**0.5
228  hiicool=de*dhp*betaf
229 
230  ! Equilibrium cooling (for high Te)
231 
232  equil=(1.0455e-18/te**0.63)*(1.-exp(-(te*1.e-5)**1.63))*de*den+hiicool
233 
234  ! switch between non-equil. and equil. ionization cooling
235 
236  IF(te.LE.44770.) fr=0.
237  IF(te.GE.54770.) fr=1.
238  IF(te.GT.44770..AND.te.LT.54770.) THEN
239  ex2=exp(-2.*(te-49770.)/500.)
240  tanh=(1.-ex2)/(1.+ex2)
241  fr=0.5*(1.+tanh)
242  END IF
243 
244  aloss=ecoll+eion+(erec+7.033*(eoi+eoii))*(1.-fr)+equil*fr
245  !
246 END FUNCTION aloss

267  real, dimension(neq) :: prim
268  real :: T !, radphi0, psi0
269  !real (kind=16) :: etau, psi, dh, y0, fpn, g0, e, y1, t1,dh0, al
270  !real (kind=16) :: gain, tprime, ce, ALOSS
271  real (kind=8) :: etau, dh, y0, g0, e, y1, t1,dh0, al
272  real (kind=8) :: tprime, ce !, ALOSS
273  real(kind=8) :: fpn, gain

277

280  !
281  ! parameters
282  ! xi - neutral carbon abundance (for non-zero electron density
283  ! boltzm - Boltzmann's constant
284  !
285  ! real (kind=8), parameter :: xi=1.d-4,boltzm=1.3807d-16,AMASS=2.158d-24
286  real (kind=8), parameter :: xi=1.d-4,boltzm=1.3807d-16,amass=2.158d-24
287 
288  ! the following is not used, and kept only to avoid compiler warnings
289  !ph0=0.
290  !psi0=0.

293  !!! radphi=radphi0*etau
294  !psi=psi0*etau
295 
296  ! solve for the ionization fraction and the internal energy
297 
298  call u2prim(uu,prim,t) !# temperature
299  col=colf(real(t,8)) !# collisional ionization rate
300  rec=alpha(real(t,8)) !# rad. recombination rate
301  y0=real( uu(neqdyn+1)/uu(1), 8 ) !# neutral H fraction
302  dh=real( uu(1), 8 ) !# H density
303  fpn=real(radphi, 8)/dh !# ionizing flux per nucleus
304  !print*,fpn
305  !fpn=0.
306 
307  ! solve for the new neutral fraction using the analytical
308  ! solution (see notes)
309 
310  a=rec+col
311 
312  if (dif_rad) then
313  b=-((2.+xi)*rec+(1.+xi)*col+fpn)
314  else
315  b=-((2.+xi)*rec+(1.+xi)*col )
316  end if
317 
318  c=(1.+xi)*rec
319  d=sqrt(b**2-4.*a*c)
320  g0=(2.*a*y0+b+d)/(2.*a*y0+b-d)
321  e=exp( -d*dh*real(dt,8) )
322 
323  y1=(-b-d*(1.+g0*e)/(1.-g0*e))/(2.*a) !# the new neutral fraction
324  y1=min(y1,0.9999)
325  y1=max(y1,0.)
326 
327  ! find the new total energy using the cooling and heating rates
328  ! and assuming that the cooling goes approximately linear with
329  ! temperature
330 
331  dh0=real( uu(neqdyn+1), 8 )
332  al=aloss(y0,y1,dt,dh,dh0,real(t,8))/dh**2
333 
334  !if(al.lt.0.) write(*,*) 'que paso !'
335  !if(al.lt.0.) al=0.
336  ! al=al*(1.-(0.5e4/max(1.e4,t))**4)
337  !if(t.le.1.e4) al=al*real((t/1.e4,8)**4)
338 
339  if (dif_rad) then
340  gain=real(radphi,8)*dh0*boltzm*1.E4 !3.14d5
341  tprime=max( gain*real(t,8)/(dh**2*al),1000.)
342  else
343  tprime=10.
344  end if
345  !tprime=1000.
346 
347  ce=(2.*dh*al)/(3.*boltzm*real(t,8))
348  t1=tprime+(t-tprime)*exp(-ce*dt) !# new temperature
349 
350  t1=max(t1,0.1*real(t,8) )
351  t1=min(t1,10.*real(t,8) )
352  ! t1=max(t1,tprime)
353 
354 !#if TWOTEMP
355 ! t1=1.E4-9990.*y1
356 !#endif
357  ! update the uu array
358  uu(neqdyn+1)=real(y1)*uu(1)
359 
360  if (mhd) then
361 #ifdef BFIELD
362  uu(5) = cv*(2.*uu(1)-uu(neqdyn+1))*real(t1)/Tempsc &
363  +0.5*prim(1)*(prim(2)**2+prim(3)**2+prim(4)**2) &
364  +0.5* (prim(6)**2+prim(7)**2+prim(8)**2)
365 #endif
366  else
367  uu(5) = cv*(2.*uu(1)-uu(neqdyn+1))*real(t1)/Tempsc &
