59 subroutine init_exo()
60 
61  use constants, only : msun, yr, ggrav, pi, mjup, au, day,rjup, rsun
62 
63  implicit none
64  real :: amdot ! mdot_star (MSUN/yr)
65  real :: ampdot ! mdot_planet (g/s)
66 
67  !----------------STAR PARAMETERS ------------------
68  masss = 1.1*msun
69  rss = 1.2*rsun
70  amdot = 2.e-14*msun/yr ! Stellar Mass Loss rate (g s^-1)
71  tsw = 1.56e6 !************* ! Stellar temperature (K)
72  ! Stellar wind, imposed at the 1.5x sonic point (cm)
73  rsw = 1.2*rsun !*************
74  vsw = 1.e5 !************* ! Stellar wind velocity (cm/s)
75  dsw = ((amdot/rsw)/(4*pi*rsw*vsw)) ! Stellar density @RS (g cm^-3)
76  bsw = 1.0 ! Stellar magnetic field (g)
77 
78  !----------------PLANET PARAMETERS------------------
79  massp = 0.67*mjup
80  ampdot= 1.e10 !*********** ! Planetary Mass Loss rate (g/s)
81  tpw = 1e4 ! Planets temperature
82  rpw = 3.*1.38*rjup ! Planetary wind radius (cm)
83  vpw = 10.e5 ! Planets wind velocity (cm/s)
84  dpw=((ampdot/rpw)/(4*pi*rpw*vpw)) ! Planetary wind density
85  bpw = 0.04 ! Planetary magnetic field (g)
86 
87  !ORBITAL PARAMETERS
88  rorb=.047*au!0.47**AU
89  torb=3.52*day
90 
91  ! change to code units
92  dsw=dsw/rhosc
93  vsw=vsw/sqrt(vsc2)
94  tsw=tsw/tempsc
95  rsw=rsw/rsc
96  rss=rss/rsc
97  bsw=bsw/bsc
98  bpw=bpw/bsc
99  dpw=dpw/rhosc
100  vpw=vpw/sqrt(vsc2)
101  tpw=tpw/tempsc
102  rpw=rpw/rsc
103  rorb=rorb/rsc
104  torb=torb/tsc
105 
106  omegap=2.*pi/torb !tidally locked planet
107 
108 
109  ! initial position
110  xp=rorb*cos(-25.*pi/180.)
111  yp=0.
112  zp=rorb*sin(-25.*pi/180.)
113 
114 end subroutine init_exo

125 subroutine impose_exo(u,time)
126 
127  use constants, only : pi
128  use globals, only : coords, dx, dy, dz
129  implicit none
130  real, intent(out) :: u(neq,nxmin:nxmax,nymin:nymax,nzmin:nzmax)
131  real, intent (in) :: time
132  real :: x, y, z, xpl, ypl, zpl
133  real :: velx, vely, velz, rads, dens, radp, phi
134  real :: vxorb, vyorb, vzorb, cpi
135  integer :: i,j,k
136 
137  phi=-25.*pi/180.
138 
139 
140  xp=rorb*cos(omegap*time+phi)
141  zp=rorb*sin(omegap*time+phi)
142 
143  !Orbital planetary velocity (moves in the xz-plane)
144  vxorb=-omegap*rorb*sin(omegap*time+phi)
145  vzorb= omegap*rorb*cos(omegap*time+phi)
146  vyorb=0.
147 
148  do i=nxmin,nxmax
149  do j=nymin,nymax
150  do k=nzmin,nzmax
151 
152  ! Position measured from the centre of the grid (star)
153  x=(float(i+coords(0)*nx-nxtot/2)+0.5)*dx
154  y=(float(j+coords(1)*ny-nytot/2)+0.5)*dy
155  z=(float(k+coords(2)*nz-nztot/2)+0.5)*dz
156 
157  ! Position measured from the centre of the planet
158  xpl=x-xp
159  ypl=y
160  zpl=z-zp
161 
162  ! Distance from the centre of the star
163  rads=sqrt(x**2+y**2+z**2)
164 
165  ! Distance from the centre of the planet
166  radp=sqrt(xpl**2+ypl**2+zpl**2)
167 
168  ! IF INSIDE THE STAR
169  if( rads <= rsw) then
170  if(rads == 0.) rads=dx*0.10
171 
172  velx=vsw*x/rads
173  vely=vsw*y/rads
174  velz=vsw*z/rads
175  dens=dsw!*RSW**2/RADS**2
176  ! total density and momena
177  u(1,i,j,k) = dens
178  u(2,i,j,k) = dens*velx
179  u(3,i,j,k) = dens*vely
180  u(4,i,j,k) = dens*velz
181  ! magnetic field
182 
183  if (pmhd .or. mhd) then
184  cpi = bsw*(rsw/rads)**3/(2.*rads**2)
185  u(6,i,j,k) = 3.*y*x*cpi
186  u(7,i,j,k) = (3.*y**2-rads**2)*cpi
187  u(8,i,j,k) = 3.*y*z*cpi
188  end if
189 
190  if (mhd) then
191  ! total energy
192  u(5,i,j,k)=0.5*dens*(velx**2+vely**2+velz**2) &
193  + cv*dens*tsw &
194  + 0.5*(u(6,i,j,k)**2+u(7,i,j,k)**2+u(8,i,j,k)**2)
195  else
196  ! total energy
197  u(5,i,j,k)=0.5*dens*(velx**2+vely**2+velz**2) &
198  + cv*dens*1.9999*tsw
199  endif
200 
201  if (passives) then
202  ! density of neutrals
203  u(neqdyn+1,i,j,k)= 0.0001*dens
204  ! passive scalar (h-hot, c-cold, i-ionized, n-neutral)
205  u(neqdyn+2,i,j,k)= dens ! passive scalar
206  ! u(neqdyn+3,i,j,k)= 0.00001*dens ! xci*rho
207  ! u(neqdyn+4,i,j,k)= 0.00001*dens ! xhn*rho
208  ! u(neqdyn+5,i,j,k)= 0.00001*dens ! xcn*rho
209  end if
210  ! IF INSIDE THE PLANET
211  !!! else if(radp <= rpw) then
212  !!!
213  !!! if(radp == 0.) radp=dx*0.10
214  !!!
215  !!! VelX=VXORB+VPW*XPL/RADP
216  !!! VelY=VYORB+VPW*YPL/RADP
217  !!! VelZ=VZORB+VPW*ZPL/RADP
218  !!! DENS=DPW!*RPW**2/RADP**2
219  !!! ! total density and momenta
220  !!! u(1,i,j,k) = dens
221  !!! u(2,i,j,k) = dens*velx
222  !!! u(3,i,j,k) = dens*vely
223  !!! u(4,i,j,k) = dens*velz
224  !!! ! Magnetic fields (IF MHD or PMHD)
225  !!!#if defined(PMHD) || defined(MHD)
226  !!! cpi = bpw*(rpw/radp)**3/(2.*radp**2)
227  !!! u(6,i,j,k) = 3.*ypl*xpl*cpi
228  !!! u(7,i,j,k) = (3.*ypl**2-radp**2)*cpi
229  !!! u(8,i,j,k) = 3.*ypl*zpl*cpi
230  !!!#endif
231  !!! ! energy
232  !!!#ifdef MHD
233  !!! u(5,i,j,k)=0.5*dens*(velx**2+vely**2+velz**2) &
234  !!! + cv*dens*1.8*Tpw &
235  !!! + 0.5*(u(6,i,j,k)**2+u(7,i,j,k)**2+u(8,i,j,k)**2)
236  !!!#else
237  !!! u(5,i,j,k)=0.5*dens*(velx**2+vely**2+velz**2) &
238  !!! + cv*dens*1.8*Tpw
239  !!!#endif
240  !!!
241  !!!#ifdef PASSIVES
242  !!! ! density of neutrals
243  !!! u(neqdyn+1,i,j,k)=0.2*dens
244  !!! ! passive scalar (h-hot, c-cold, i-ionized, n-neutral)
245  !!! u(neqdyn+2,i,j,k)= -dens ! passive scalar
246  !!!
247  !!!! u(neqdyn+3,i,j,k)= 0.00001*dens ! xci*rho
248  !!!! u(neqdyn+4,i,j,k)= 0.00001*dens ! xhn*rho
249  !!!! u(neqdyn+5,i,j,k)= 0.99999*dens ! xcn*rho
250  !!!#endif
251  end if
252 
253  end do
254  end do
255 end do
256 
257 end subroutine impose_exo
