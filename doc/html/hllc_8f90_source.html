44 subroutine prim2fhllc(priml,primr,ff)
45 
46  use parameters, only : neq, neqdyn, cv, pmhd, passives
47  use hydro_core, only : csound, prim2f, prim2u
48  implicit none
49  real, dimension(neq),intent(in ) :: priml, primr
50  real, dimension(neq),intent(inout) :: ff
51  real, dimension(neq) :: uu, uuk
52  real :: csl, csr, sl, sr, slmul, srmur, rholul, rhorur, sst
53  real :: rhost,ek
54 
55  call csound(priml(5),priml(1),csl)
56  call csound(primr(5),primr(1),csr)
57 
58  sr=max(priml(2)+csl,primr(2)+csr)
59  sl=min(priml(2)-csl,primr(2)-csr)
60 
61  if (sl > 0) then
62  call prim2f(priml,ff)
63  return
64  endif
65 
66  if (sr < 0) then
67  call prim2f(primr,ff)
68  return
69  endif
70 
71  slmul=sl-priml(2)
72  srmur=sr-primr(2)
73  rholul=priml(1)*priml(2)
74  rhorur=primr(1)*primr(2)
75 
76  sst = (srmur*rhorur-slmul*rholul-primr(5)+priml(5) ) &
77  / (srmur*primr(1)-slmul*priml(1) )
78 
79  if (sst >= 0.) then
80  rhost=priml(1)*(slmul)/(sl-sst)
81  ek= 0.5*priml(1)*(priml(2)**2+priml(3)**2+priml(4)**2)+cv*priml(5)
82 
83  uuk(1)=rhost
84  uuk(2)=rhost*sst
85  uuk(3)=rhost*priml(3)
86  uuk(4)=rhost*priml(4)
87  uuk(5)=rhost*( ek/priml(1)+(sst-priml(2))*(sst+priml(5)/(priml(1)*slmul)) )
88 
89  if (pmhd) then
90  uuk(6:8)=rhost*priml(6:8)/priml(1)
91  end if
92  if (passives) then
93  uuk(neqdyn+1:neq)=rhost*priml(neqdyn+1:neq)/priml(1)
94  end if
95 
96  call prim2f(priml,ff)
97  call prim2u(priml,uu)
98  ff(:)=ff(:) + sl*( uuk(:)-uu(:) )
99  return
100  endif
101 
102  if (sst <= 0.) then
103  rhost=primr(1)*(srmur)/(sr-sst)
104  ek= 0.5*primr(1)*(primr(2)**2+primr(3)**2+primr(4)**2)+cv*primr(5)
105 
106  uuk(1)=rhost
107  uuk(2)=rhost*sst
108  uuk(3)=rhost*primr(3)
109  uuk(4)=rhost*primr(4)
110  uuk(5)=rhost*( ek/primr(1)+(sst-primr(2))*(sst+primr(5)/(primr(1)*srmur)) )
111 
112  if (pmhd) then
113  !uuk(5)= 0.
114  uuk(6:8)=rhost*primr(6:8)/primr(1)
115  end if
116  if (passives) then
117  uuk(neqdyn+1:neq)=rhost*primr(neqdyn+1:neq)/primr(1)
118  end if
119 
120  call prim2f(primr,ff)
121  call prim2u(primr,uu)
122  ff(:)=ff(:) + sr*( uuk(:)-uu(:) )
123  return
124  endif
125 
126  print*, 'Error in hllc'
127  print*, 'primL: ',priml(:)
128  print*, 'primR: ',primr(:)
129  stop

152  !

184  call prim2fhllc(priml,primr,ff)
185  call swapz(ff,neq)
186  h(:,i,j,k)=ff(:)
187 
188  end do
189  end do
190  end do
191 
192  case (2) ! 2nd half timestep
193 
194  do k=0,nz
195  do j=0,ny
196  do i=0,nx
197 
198  !------- x direction ------------------------------------
199  priml(:)=primit(:,i, j,k )
200  primr(:)=primit(:,i+1,j,k )
201  primll(:)=primit(:,i-1,j,k )
202  primrr(:)=primit(:,i+2,j,k )

207

210  primr(:)=primit(:,i,j+1,k )
211  primll(:)=primit(:,i,j-1,k )
212  primrr(:)=primit(:,i,j+2,k )
213  call swapy(priml,neq)
214  call swapy(primr,neq)
215  call swapy(primll,neq)
216  call swapy(primrr,neq)
217  call limiter(primll,priml,primr,primrr,neq)
218 
219  call prim2fhllc(priml,primr,ff)
220  call swapy(ff,neq)
221  g(:,i,j,k)=ff(:)
222 
223  !------- z direction ------------------------------------

225  primr(:)=primit(:,i,j,k+1 )
226  primll(:)=primit(:,i,j,k-1 )
227  primrr(:)=primit(:,i,j,k+2)
228  call swapz(priml,neq)
229  call swapz(primr,neq)
230  call swapz(primll,neq)

233

235  call swapz(ff,neq)
236  h(:,i,j,k)=ff(:)
237 
238  end do

242  end select
