114 subroutine coolingchi()
115 
116  use parameters, only : nx, ny, nz, cv, psc, tsc
117  use globals, only : u, primit, dt_cfl
118  use hydro_core, only : u2prim
119  implicit none

121  real, parameter :: Tmin=10000.
122  real (kind=8) :: ALOSS, Ce
123  integer :: i, j, k
124  real :: dt_seconds
125 
126  dt_seconds = dt_cfl*tsc
127 
128  do k=1,nz
129  do j=1,ny
130  do i=1,nx
131 
132  ! get the primitives (and T)
133  call u2prim(u(:,i,j,k),primit(:,i,j,k),t)
134 
135  if(t > tmin) then
136 
137  eth0=cv*primit(5,i,j,k)
138 
139  aloss=coolchi(t)
140  dens=primit(1,i,j,k)
141  ce=(aloss*dble(dens)**2)/(eth0*psc) ! cgs
142 
143  ! apply cooling to primitive and conserved variables
144  primit(5,i,j,k)=primit(5,i,j,k)*exp(-ce*dt_seconds)
145
