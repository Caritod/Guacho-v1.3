82 subroutine grav_source(xc,yc,zc,pp,s)
83  use constants, only : ggrav
84  use exoplanet ! this module contains the position of the planet
85  ! to do: uncouple from exoplanet module, at this moment to make
86  ! a dirty fix I'm placing a copy of the exoplanet to the general
87  ! source tree
88  implicit none
89  real, intent(in) :: xc, yc, zc
90  real, intent(in) :: pp(neq)
91  real, intent(inout) :: s(neq)
92  integer, parameter :: nb=2 ! 2 particles
93  real :: x(nb),y(nb),z(nb), GM(nb), rad2(nb)
94  integer :: i
95 
96  gm(1)=0.3*ggrav*masss/rsc/vsc2
97  gm(2)=ggrav*massp/rsc/vsc2
98 
99  !calculate distance from the sources
100  ! star
101  x(1)=xc
102  y(1)=yc
103  z(1)=zc
104  rad2(1) = x(1)**2 +y(1)**2 + z(1)**2
105  ! planet
106  x(2)=xc-xp
107  y(2)=yc
108  z(2)=zc-zp
109  rad2(2) = x(2)**2 +y(2)**2 + z(2)**2
110 
111  ! update source terms
112  do i=1, nb
113  ! momenta
114  s(2)= s(2)-pp(1)*gm(i)*x(i)/(rad2(i)**1.5)
115  s(3)= s(3)-pp(1)*gm(i)*y(i)/(rad2(i)**1.5)
116  s(4)= s(4)-pp(1)*gm(i)*z(i)/(rad2(i)**1.5)
117  ! energy
118  s(5)= s(5)-pp(1)*gm(i)*( pp(2)*x(i) +pp(3)*y(i) +pp(4)*z(i) ) &
119  /(rad2(i)**1.5 )
120  end do
121 
122 
123 end subroutine grav_source

139 subroutine radpress_source(i,j,k,xc,yc,zc,rc,pp,s)
140 
141  use difrad
142  implicit none
143  integer, intent(in) :: i,j,k
144  real, intent(in) :: xc, yc, zc, rc, pp(neq)
145  real, intent(inout) :: s(neq)
146  real :: radphi
147  ! the following is h/912Angstroms = h/lambda
148  real :: hlambda= 7.265e-22, frad
149 
150  radphi= ph(i,j,k)
151 
152  ! update the source terms
153  frad=radphi*pp(6)*hlambda*rsc/rhosc/vsc2
154  ! momenta
155  s(2) = s(2) + frad*xc/rc
156  s(3) = s(3) + frad*yc/rc
157  s(4) = s(4) + frad*zc/rc
158  ! energy
159  s(5) = s(5)+ frad*( xc*pp(2) + yc*pp(3) + zc*pp(4) )/rc
160 
161 end subroutine radpress_source

232 subroutine source(i,j,k,prim,s)
233 
234  implicit none
235  integer, intent(in) :: i, j, k
236  real, intent(in) :: prim(neq)
237  real, intent(out) :: s(neq)
238  real :: x, y, z, r
239 
240  ! resets the source terms
241  s(:) = 0.
242 
243  ! position with respect to the center of the grid
244  call getpos( i, j, k, x, y ,z, r)
245 
246  ! point source(s) gravity
247  if (point_grav) call grav_source(x,y,z,prim,s)
248 
249  ! photoionization radiation pressure
250  if (radiation_pressure) call radpress_source(i,j,k,x,y,z,r,prim,s)
251 
252  ! divergence correction Powell et al. 1999
253  if (eight_wave) call divbcorr_8w_source(i,j,k,prim,s)
254  return
