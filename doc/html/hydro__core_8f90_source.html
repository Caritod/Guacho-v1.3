44 subroutine u2prim(uu, prim, T)
45

47  pmhd, mhd, eq_of_state

49  use network, only : n_spec

52  real, intent(out), dimension(neq) :: prim
53  real, intent(out) :: T
54  real :: r
55  integer :: i

58  r=max(uu(1),1e-15)
59  prim(1)=r
60  prim(2)=uu(2)/r
61  prim(3)=uu(3)/r
62  prim(4)=uu(4)/r
63

65  prim(5)=( uu(5)-0.5*r*(prim(2)**2+prim(3)**2+prim(4)**2) &
66  -0.5* ( uu(6)**2+ uu(7)**2 +uu(8)**2) ) /cv
67  else
68  prim(5)=( uu(5)-0.5*r*(prim(2)**2+prim(3)**2+prim(4)**2) ) /cv
69  end if
70

72

74  prim(6:8) = uu(6:8)
75  end if
76 
77  if (passives) then
78  prim(neqdyn+1:neq) = uu(neqdyn+1:neq)
79  end if
80 
81  ! Temperature calculation
82 
83  if (eq_of_state == eos_adiabatic) then
84  t=(prim(5)/r)*tempsc
85  end if
86 
87  if (eq_of_state == eos_single_specie) then
88  ! assumes it is fully ionized
89  r=max(r,1e-15)
90  t=max(1.,(prim(5)/r)*tempsc)
91  prim(5)=r*t/tempsc
92  end if
93 
94  if (eq_of_state == eos_h_rate) then
95  dentot=(2.*r-prim(neqdyn+1))
96  dentot=max(dentot,1e-15)
97  t=max(1.,(prim(5)/dentot)*tempsc)
98  prim(5)=dentot*t/tempsc
99  end if
100 
101  if (eq_of_state == eos_chem) then
102  ! Assumes that rho scaling is mu*mh
103  dentot = 0.
104  do i = neqdyn+1, neqdyn+n_spec
105  dentot = prim(i) + dentot
106  end do
107  dentot = max(dentot, 1e-15)
108  t=max(1.,(prim(5)/dentot)*tempsc )
109 
110  !T=(prim(5)/r)*Tempsc
111  prim(5) = dentot * t /tempsc
112  end if

115 end subroutine u2prim

259  + cv*prim(5)
260

263  ff(3) = prim(1)*prim(2)*prim(3)-prim(6)*prim(7)
264  ff(4) = prim(1)*prim(2)*prim(4)-prim(6)*prim(8)
265  ff(5) = prim(2)*(etot+prim(5)+0.5*(prim(6)**2+prim(7)**2+prim(8)**2) ) &
266  -prim(6)*(prim(2)*prim(6)+prim(3)*prim(7)+prim(4)*prim(8))
267  else
268  ! HD or PMHD
269  etot= 0.5*prim(1)*(prim(2)**2+prim(3)**2+prim(4)**2)+cv*prim(5)
270

272  ff(2) = prim(1)*prim(2)*prim(2)+prim(5)
273  ff(3) = prim(1)*prim(2)*prim(3)
274  ff(4) = prim(1)*prim(2)*prim(4)
275  ff(5) = prim(2)*(etot+prim(5))

427 subroutine get_timestep(current_iter, n_iter, current_time, tprint, dt, dump_flag)
428 
429  use parameters, only : nx, ny, nz, cfl, mpi_real_kind, mhd
430  use globals, only : primit, dx, dy, dz
431  implicit none
432 #ifdef MPIP
433  include "mpif.h"
434 #endif

436  real, intent(in) :: current_time, tprint
437  real, intent(out) :: dt

439  real :: dtp
440  real :: c, cx, cy, cz
441  integer :: i, j, k, err
442 
443  dtp=1.e30
444 
445  do k=1,nz
446  do j=1,ny
447  do i=1,nx
448 
449  if (mhd) then
450  call cfast(primit(5,i,j,k),primit(1,i,j,k),&
451  primit(6,i,j,k), primit(7,i,j,k), primit(8,i,j,k), &
452  cx,cy,cz)
453  dtp=min(dtp,dx/(abs(primit(2,i,j,k))+cx))
454  dtp=min(dtp,dy/(abs(primit(3,i,j,k))+cy))
455  dtp=min(dtp,dz/(abs(primit(4,i,j,k))+cz))
456  else
457  call csound(primit(5,i,j,k),primit(1,i,j,k),c)
458  dtp=min(dtp,dx/(abs(primit(2,i,j,k))+c))
459  dtp=min(dtp,dy/(abs(primit(3,i,j,k))+c))
460  dtp=min(dtp,dz/(abs(primit(4,i,j,k))+c))
461  end if
462 
463  end do
464  end do
465  end do
466 
467  if (current_iter <= n_iter ) then
468  ! boots up simulation with small CFL
469  dtp = cfl*(2.**(-real(n_iter+1-current_iter)))*dtp
470  else
471  dtp=cfl*dtp
472  end if
473 
474 #ifdef MPIP
475  call mpi_allreduce(dtp, dt, 1, mpi_real_kind, mpi_min, mpi_comm_world,err)
476 #else
477  dt=dtp
478 #endif
479 
480  ! Adjust dt so t+dt doesnt overshoot tprint
481  if(( current_time + dt )>=tprint) then
482  dt= tprint-current_time
483  dump_flag=.true.
484  endif
485 
486 
487 end subroutine get_timestep

525  real function average(a,b)
526  use constants
527  use parameters, only : slope_limiter
528  implicit none
529  real, intent(in) :: a, b
530 
531  If (slope_limiter == limiter_no_average) then
532  ! no average (reduces to 1st order)
533  average=0.
534  end if
535 
536  if (slope_limiter == limiter_no_limit) then
537  ! no limiter
538  average=0.5*(a+b)
539  end if
540 
541  if (slope_limiter == limiter_minmod) then
542  ! Minmod - most diffusive
543  s=sign(1.,a)
544  average=s*max(0.,min(abs(a),s*b))
545  end if
546 
547  if (slope_limiter == limiter_van_leer) then
548  ! Falle Limiter (Van Leer)
549  if(a*b.le.0.) then
550  average=0.
551  else
552  average=a*b*(a+b)/(a**2+b**2)
553  end if
554  end if
555 
556  if (slope_limiter == limiter_van_albada) then
557  ! Van Albada
558  average=(a*(b*b+delta)+b*(a*a+delta))/(a*a+b*b+delta)
559  end if
560 
561  if (slope_limiter == limiter_umist) then
562  ! UMIST Limiter - less diffusive
563  s=sign(1.,a)
564  c=0.25*a+0.75*b
565  d=0.75*a+0.25*b
566  average=min(2.*abs(a),2.*s*b,s*c,s*d)
567  average=s*max(0.,average)
568  end if
569 
570  if (slope_limiter == limiter_woodward) then
571  ! Woodward Limiter (o MC-limiter; monotonized central difference)
572  s=sign(1.,a)
573  c=0.5*(a+b)
574  average=min(2.*abs(a),2.*s*b,s*c)
575  average=s*max(0.,average)
576  end if
577 
578  if (slope_limiter == limiter_superbee) then
579  ! superbee Limiter (tends to flatten circular waves)
580  s=sign(1.,b)
581  av1=min(2.*abs(b),s*a)
582  av2=min(abs(b),2.*s*a)
583  average=s*max(0.,av1,av2)
584  end if
585 
586  end function average
587 
588 end subroutine limiter
