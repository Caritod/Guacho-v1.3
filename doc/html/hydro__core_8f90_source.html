46 subroutine u2prim(uu, prim, T)
47 
48  use parameters, only : neq, neqdyn, tempsc, vsc2, cv, passives, &
49  pmhd, mhd, eq_of_state
50  use constants
51  use network, only : n_spec
52  implicit none
53  real, intent(in), dimension(neq) :: uu
54  real, intent(out), dimension(neq) :: prim
55  real, intent(out) :: T
56  real :: r
57 #ifdef PASSIVES
58  integer :: i
59  real :: dentot
60 #endif
61 
62  r=max(uu(1),1e-15)
63  prim(1)=r
64 
65  prim(2)=uu(2)/r
66  prim(3)=uu(3)/r
67  prim(4)=uu(4)/r
68 
69  if (mhd) then
70 #ifdef BFIELD
71  prim(5)=( uu(5)-0.5*r*(prim(2)**2+prim(3)**2+prim(4)**2) &
72  -0.5* ( uu(6)**2+ uu(7)**2 +uu(8)**2) ) /cv
73 #endif
74  else
75  prim(5)=( uu(5)-0.5*r*(prim(2)**2+prim(3)**2+prim(4)**2) ) /cv
76  end if
77 
78  prim(5)=max(prim(5),1e-16)
79 
80 #ifdef BFIELD
81  if (mhd .or. pmhd) then
82  prim(6:8) = uu(6:8)
83  end if
84 #endif
85 
86 #ifdef PASSIVES
87  if (passives) then
88  prim(neqdyn+1:neq) = uu(neqdyn+1:neq)
89  end if
90 #endif
91 
92  ! Temperature calculation
93 
94  if (eq_of_state == eos_adiabatic) then
95  t=(prim(5)/r)*tempsc
96  end if
97 
98  if (eq_of_state == eos_single_specie) then
99  ! assumes it is fully ionized
100  r=max(r,1e-15)
101  t=max(1.,(prim(5)/r)*tempsc)
102  prim(5)=r*t/tempsc
103  end if
104 
105 
106 #ifdef PASSIVES
107 
108  if (eq_of_state == eos_h_rate) then
109  dentot=(2.*r-prim(neqdyn+1))
110  dentot=max(dentot,1e-15)
111  t=max(1.,(prim(5)/dentot)*tempsc)
112  prim(5)=dentot*t/tempsc
113  end if
114 
115  if (eq_of_state == eos_chem) then
116  ! Assumes that rho scaling is mu*mh
117  dentot = 0.
118  do i = neqdyn+1, neqdyn+n_spec
119  dentot = prim(i) + dentot
120  end do
121  dentot = max(dentot, 1e-15)
122  t=max(1.,(prim(5)/dentot)*tempsc )
123 
124  !T=(prim(5)/r)*Tempsc
125  prim(5) = dentot * t /tempsc
126  end if
127 
128 #endif
129 
130 end subroutine u2prim

246 subroutine calcprim(u,primit, only_ghost)
247 
248  use parameters, only : neq, nxmin, nxmax, nymin, nymax, nzmin, nzmax, &
249  nx, ny, nz, riemann_solver
250  use constants
251 
252  use globals, only : temp, primit0
253  implicit none

258 
259  if (present(only_ghost) .and. only_ghost) then

261  ! k = 0, and, nz

271  end if
272  end do
273  end do
274 
275  ! j = 0, and, ny

277  do i=0,nx+1
278  if (riemann_solver == solver_hlle_split_all .or. &
279  riemann_solver == solver_hlld_split_all) then

282  else
283  call u2prim(u(:,i,0 ,k),primit(:,i,0 ,k),temp(i,0 ,k) )
284  call u2prim(u(:,i,ny+1,k),primit(:,i,ny+1,k),temp(i,ny+1,k) )
285  end if
286  end do
287  end do
288 
289  ! i = 0, and, nx
290  do k=0,nz+1
291  do j=0,ny+1
292  if (riemann_solver == solver_hlle_split_all .or. &
293  riemann_solver == solver_hlld_split_all) then
294  call u2primsplitall(u(:,0 ,j,k),primit(:,0 ,j,k),primit0(:,0 ,j,k),temp(i,0 ,k) )
295  call u2primsplitall(u(:,nx+1,j,k),primit(:,nx+1,j,k),primit0(:,nx+1,j,k),temp(nx+1,j,k) )
296  else
297  call u2prim(u(:,0 ,j,k),primit(:,0 ,j,k),temp(0 ,j,k) )
298  call u2prim(u(:,nx+1,j,k),primit(:,nx+1,j,k),temp(nx+1,j,k) )
299  end if
300  end do
301  end do
302  !-----------------------
303  else
304  ! entire domain
305  do k=nzmin,nzmax
306  do j=nymin,nymax
307  do i=nxmin,nxmax
308  if (riemann_solver == solver_hlle_split_all .or. &
309  riemann_solver == solver_hlld_split_all) then
310  call u2primsplitall(u(:,i,j,k),primit(:,i,j,k),primit0(:,i,j,k),temp(i,j,k) )
311  else
312  call u2prim(u(:,i,j,k),primit(:,i,j,k),temp(i,j,k) )
313  end if
314 
315  end do
316  end do
317  end do
318 
319  end if

321 end subroutine calcprim

403 subroutine prim2f(prim,ff,prim0)
404  use parameters, only : neq, neqdyn, cv, pmhd, mhd, passives, riemann_solver
405  use constants, only : solver_hlle_split_all, solver_hlld_split_all
406  implicit none
407  real, dimension(neq), intent(in) :: prim
408  real, dimension(neq), intent(in), optional :: prim0
409  real, dimension(neq), intent(out) :: ff
410  real :: etot
411 
412 
413  if (mhd) then
414 #ifdef BFIELD
415  if (riemann_solver == solver_hlle_split_all .or. &
416  riemann_solver == solver_hlld_split_all) then
417  if (present(prim0)) then
418  etot = 0.5*( (prim(1)+prim0(1))*(prim(2)**2+prim(3)**2+prim(4)**2) &
419  + prim(6)**2+prim(7)**2+prim(8)**2 ) &
420  + cv*prim(5) &
421  + prim0(6)*prim(6)+prim0(7)*prim(7)+prim0(8)*prim(8)
422 
423  ff(1) = (prim(1)+prim0(1))*prim(2)
424  ff(2) = (prim(1)+prim0(1))*prim(2)*prim(2)+prim(5)+0.5*(prim(7)**2+prim(8)**2-prim(6)**2) &
425  - prim0(6)*prim(6)+prim0(7)*prim(7)+prim0(8)*prim(8)
426  ff(3) = (prim(1)+prim0(1))*prim(2)*prim(3)-prim(6)*prim(7) &
427  - prim0(7)*prim(6)-prim0(6)*prim(7)
428  ff(4) = (prim(1)+prim0(1))*prim(2)*prim(4)-prim(6)*prim(8) &
429  - prim0(8)*prim(6)-prim0(6)*prim(8)
430  ff(5) = prim(2)*(etot+prim(5)+0.5*(prim(6)**2+prim(7)**2+prim(8)**2) &
431  +cv*prim0(5)+0.5*(prim0(6)**2+prim0(7)**2+prim0(8)**2) ) &
432  -(prim(6)+prim0(6))*(prim(2)*(prim(6)+prim0(6))+ &
433  prim(3)*(prim(7)+prim0(7))+prim(4)*(prim(8)+prim0(8))) !REVISAR
434  end if
435  else
436  ! MHD (active)
437  etot= 0.5*( prim(1)*(prim(2)**2+prim(3)**2+prim(4)**2) &
438  + prim(6)**2+prim(7)**2+prim(8)**2 ) &
439  + cv*prim(5)
440 
441  ff(1) = prim(1)*prim(2)
442  ff(2) = prim(1)*prim(2)*prim(2)+prim(5)+0.5*(prim(7)**2+prim(8)**2-prim(6)**2)
443  ff(3) = prim(1)*prim(2)*prim(3)-prim(6)*prim(7)
444  ff(4) = prim(1)*prim(2)*prim(4)-prim(6)*prim(8)
445  ff(5) = prim(2)*(etot+prim(5)+0.5*(prim(6)**2+prim(7)**2+prim(8)**2) ) &
446  -prim(6)*(prim(2)*prim(6)+prim(3)*prim(7)+prim(4)*prim(8))
447  end if
448 #endif /* MHD */
449  else
450  ! HD or PMHD
451  etot= 0.5*prim(1)*(prim(2)**2+prim(3)**2+prim(4)**2)+cv*prim(5)
452 
453  ff(1) = prim(1)*prim(2)
454  ff(2) = prim(1)*prim(2)*prim(2)+prim(5)
455  ff(3) = prim(1)*prim(2)*prim(3)
456  ff(4) = prim(1)*prim(2)*prim(4)
457  ff(5) = prim(2)*(etot+prim(5))
458  end if
459 
460 #ifdef BFIELD
461  if (mhd .or. pmhd) then
462  ff(6)=0.0
463  ff(7)=prim(2)*prim(7)-prim(6)*prim(3)
464  ff(8)=prim(2)*prim(8)-prim(6)*prim(4)
465  end if
466 #endif
467 
468 #ifdef PASSIVES
469  if (passives) then
470  ff(neqdyn+1:neq) = prim(neqdyn+1:neq)*prim(2)
471  end if
472 #endif
473 
474 end subroutine prim2f

617 subroutine get_timestep(current_iter, n_iter, current_time, tprint, dt, dump_flag)
618  use constants, only : solver_hlle_split_all, solver_hlld_split_all
619  use parameters, only : nx, ny, nz, cfl, mpi_real_kind, mhd, riemann_solver
620  use globals, only : primit, dx, dy, dz, primit0
621  implicit none
622 #ifdef MPIP
623  include "mpif.h"
624 #endif
625  integer, intent(in) :: current_iter, n_iter
626  real, intent(in) :: current_time, tprint
627  real, intent(out) :: dt
628  logical, intent(out) :: dump_flag
629  real :: dtp
630  real :: c, cx, cy, cz
631  integer :: i, j, k, err
632 
633  dtp=1.e30
634 
635  do k=1,nz
636  do j=1,ny
637  do i=1,nx
638 
639  if (mhd) then
640 #ifdef BFIELD
641  if (riemann_solver == solver_hlle_split_all .or. &
642  riemann_solver == solver_hlld_split_all) then
643  call cfast(primit(5,i,j,k)+primit0(5,i,j,k),primit(1,i,j,k)+primit0(1,i,j,k),&
644  primit(6,i,j,k)+primit0(6,i,j,k), primit(7,i,j,k)+primit0(7,i,j,k), primit(8,i,j,k)+primit0(8,i,j,k), &
645  cx,cy,cz)
646  else
647  call cfast(primit(5,i,j,k),primit(1,i,j,k),&
648  primit(6,i,j,k), primit(7,i,j,k), primit(8,i,j,k), &
649  cx,cy,cz)
650  end if
651 
652  dtp=min(dtp,dx/(abs(primit(2,i,j,k))+cx))
653  dtp=min(dtp,dy/(abs(primit(3,i,j,k))+cy))
654  dtp=min(dtp,dz/(abs(primit(4,i,j,k))+cz))
655 #endif
656  else
657  ! plain old hydro
658  call csound(primit(5,i,j,k),primit(1,i,j,k),c)
659  dtp=min(dtp,dx/(abs(primit(2,i,j,k))+c))
660  dtp=min(dtp,dy/(abs(primit(3,i,j,k))+c))
661  dtp=min(dtp,dz/(abs(primit(4,i,j,k))+c))
662  end if
663 
664  end do
665  end do
666  end do
667 
668  if (current_iter <= n_iter ) then
669  ! boots up simulation with small CFL
670  dtp = cfl*(2.**(-real(n_iter+1-current_iter)))*dtp
671  else
672  dtp=cfl*dtp
673  end if
674 
675 #ifdef MPIP
676  call mpi_allreduce(dtp, dt, 1, mpi_real_kind, mpi_min, mpi_comm_world,err)
677 #else
678  dt=dtp
679 #endif
680 
681  ! Adjust dt so t+dt doesnt overshoot tprint
682  if(( current_time + dt )>=tprint) then
683  dt= tprint-current_time
684  dump_flag=.true.
685  endif
686 
687 
688 end subroutine get_timestep

708  real, dimension(neq), intent(in) :: pll, prr
709  real :: dl, dm, dr, al, ar
710  integer :: ieq
711  real :: s, c, d, av1, av2
712  real, parameter :: delta=1.e-7
713 
714  do ieq=1,neq
715  dl=pl(ieq)-pll(ieq)
716  dm=pr(ieq)-pl(ieq)
717  dr=prr(ieq)-pr(ieq)
718  al=average(dl,dm)

726  real function average(a,b)
727  use constants
728  use parameters, only : slope_limiter
729  implicit none
730  real, intent(in) :: a, b
731 
732  If (slope_limiter == limiter_no_average) then
733  ! no average (reduces to 1st order)
734  average=0.
735  end if
736 
737  if (slope_limiter == limiter_no_limit) then
738  ! no limiter
739  average=0.5*(a+b)
740  end if
741 
742  if (slope_limiter == limiter_minmod) then
743  ! Minmod - most diffusive
744  s=sign(1.,a)
745  average=s*max(0.,min(abs(a),s*b))
746  end if
747 
748  if (slope_limiter == limiter_van_leer) then
749  ! Falle Limiter (Van Leer)
750  if(a*b.le.0.) then
751  average=0.
752  else
753  average=a*b*(a+b)/(a**2+b**2)
754  end if
755  end if
756 
757  if (slope_limiter == limiter_van_albada) then
758  ! Van Albada
759  average=(a*(b*b+delta)+b*(a*a+delta))/(a*a+b*b+delta)
760  end if
761 
762  if (slope_limiter == limiter_umist) then
763  ! UMIST Limiter - less diffusive
764  s=sign(1.,a)
765  c=0.25*a+0.75*b
766  d=0.75*a+0.25*b
767  average=min(2.*abs(a),2.*s*b,s*c,s*d)
768  average=s*max(0.,average)
769  end if
770 
771  if (slope_limiter == limiter_woodward) then
772  ! Woodward Limiter (o MC-limiter; monotonized central difference)
773  s=sign(1.,a)
774  c=0.5*(a+b)
775  average=min(2.*abs(a),2.*s*b,s*c)
776  average=s*max(0.,average)
777  end if
778 
779  if (slope_limiter == limiter_superbee) then
780  ! superbee Limiter (tends to flatten circular waves)
781  s=sign(1.,b)
782  av1=min(2.*abs(b),s*a)
783  av2=min(abs(b),2.*s*a)
784  average=s*max(0.,av1,av2)
785  end if
786 
787  end function average
