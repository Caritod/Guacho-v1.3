90 subroutine chemstep(y,y0,T, deltt)
91  use linear_system
92  use network, only : n_spec, n_reac, n_elem, get_reaction_rates, &
93  derv, get_jacobian, n_nequ, check_no_conservation

97  real (kind=8) :: dtm

99  real (kind=8) :: rate(n_reac),dydt(n_spec),jac(n_spec,n_spec)
100  integer, parameter :: niter=100 ! number of iterations
101  integer :: n,i,iff
102 
103  n=0
104  dtm=1./deltt
105  iff=1
106  yin(:) =y(:)
107  y0_in(:) = y0(:)
108 
109  call get_reaction_rates(rate,min(t,3e4))
110 
111  ! initial guess for Newton-Raphson
112  if ( check_no_conservation(y,y0_in) ) then
113  !print*, '*****Reset Initial Guess ********'
114  !print*, "T=", T
115  call nr_init(y,y0_in)
116  end if
117 
118  do while ( n <= niter )
119 
120  call derv(y,rate,dydt,y0)
121  call get_jacobian(y,jac,rate)
122 
123  do i=1,n_nequ
124  jac(i,i)=jac(i,i)-dtm
125  dydt(i)=dydt(i)-(y(i)-yin(i))*dtm
126  end do
127  y1(:)=-dydt(:)
128 
129  call linsys(jac,y1, n_spec)
130 
131  y(:)=y(:) + y1(:)
132  y(:)=max(y(:),1.e-40)
133 
134  yt(:)=y1(:)/y(:)
135 
136  ! exit the loop if converged
137  if(all(abs(y1(:)) <= 0.0001)) exit
138 
139  n=n+1
140 
141  end do

144  ! print*, "failed to converge after ", niter, " iterations"
